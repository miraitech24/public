---------- 2. ワームホール基本式 ----------
モリス・ソーン型ワームホールの計量
(%i6) b(l, b0) := b0^2 / sqrt(b0^2 + l^2); /* 形状関数 */

/* 赤方偏移関数（ゼロと仮定） */
Phi(l) := 0;

\[\]\[\tag{%o5} \mathop{b}\left( l\mathop{,}\ensuremath{\mathrm{b0}}\right) \mathop{:=}\frac{{{\ensuremath{\mathrm{b0}}}^{2}}}{\sqrt{{{\ensuremath{\mathrm{b0}}}^{2}}\mathop{+}{{l}^{2}}}}\]

\[\]\[\tag{%o6} \mathop{Phi}(l)\mathop{:=}0\]

エキゾチック物質のエネルギー密度
(%i7) rho(l, b0) := −c^4/(8·pi·G) · b0^2/(b0^2 + l^2)^2;

\[\]\[\tag{%o7} \mathop{rho}\left( l\mathop{,}\ensuremath{\mathrm{b0}}\right) \mathop{:=}\frac{\frac{\mathop{-}{{c}^{4}}}{8 pi G} {{\ensuremath{\mathrm{b0}}}^{2}}}{{{\left( {{\ensuremath{\mathrm{b0}}}^{2}}\mathop{+}{{l}^{2}}\right) }^{2}}}\]

エキゾチック物質の圧力（放射方向）
(%i8) p_r(l, b0) := −c^4/(8·pi·G) · b0^2/(b0^2 + l^2)^2;

\[\]\[\tag{%o8} {p_r}\left( l\mathop{,}\ensuremath{\mathrm{b0}}\right) \mathop{:=}\frac{\frac{\mathop{-}{{c}^{4}}}{8 pi G} {{\ensuremath{\mathrm{b0}}}^{2}}}{{{\left( {{\ensuremath{\mathrm{b0}}}^{2}}\mathop{+}{{l}^{2}}\right) }^{2}}}\]

エキゾチック物質の張力（横方向）
(%i9) p_t(l, b0) := c^4/(8·pi·G) · b0^2/(b0^2 + l^2)^2;

\[\]\[\tag{%o9} {p_t}\left( l\mathop{,}\ensuremath{\mathrm{b0}}\right) \mathop{:=}\frac{\frac{{{c}^{4}}}{8 pi G} {{\ensuremath{\mathrm{b0}}}^{2}}}{{{\left( {{\ensuremath{\mathrm{b0}}}^{2}}\mathop{+}{{l}^{2}}\right) }^{2}}}\]

エネルギー条件違反の度合い
(%i10) energy_violation(l, b0) := (rho(l, b0) + p_r(l, b0) + 2·p_t(l, b0)) / (abs(rho(l, b0)));

\[\]\[\tag{%o10} \mathop{energy\_ violation}\left( l\mathop{,}\ensuremath{\mathrm{b0}}\right) \mathop{:=}\frac{\mathop{rho}\left( l\mathop{,}\ensuremath{\mathrm{b0}}\right) \mathop{+}{p_r}\left( l\mathop{,}\ensuremath{\mathrm{b0}}\right) \mathop{+}2 {p_t}\left( l\mathop{,}\ensuremath{\mathrm{b0}}\right) }{\left| \mathop{rho}\left( l\mathop{,}\ensuremath{\mathrm{b0}}\right) \right| }\]
