3.2 移動による時間遅延（特殊相対論）
速度 v で移動する場合の時間遅延: Δt' = Δt / γ
v = c の時の処理を追加
(%i14) lorentz_factor(v) := block(
    [ratio],
    ratio: v^2/c^2,
    if abs(1 − ratio) < 1e-10 then 1e10 /* v ≈ c の時は大きな値 */
    else 1/sqrt(1 − ratio)
);

\[\]\[\tag{%o14} \mathop{lorentz\_ factor}(v)\mathop{:=}\mathop{block}\left( \left[ \ensuremath{\mathrm{ratio}}\right] \mathop{,}\ensuremath{\mathrm{ratio}}\mathop{:}\frac{{{v}^{2}}}{{{c}^{2}}}\mathop{,}\mathop{if}\mathop{ }\left| 1\mathop{-}\ensuremath{\mathrm{ratio}}\right| \mathop{&lt; }1.0 {{10}^{-10}}\mathop{ }\mathop{then}\mathop{ }1.0 {{10}^{10}}\mathop{ }\mathop{else}\mathop{ }\frac{1}{\sqrt{1\mathop{-}\ensuremath{\mathrm{ratio}}}}\right) \]

(%i22) delta_t_timemachine(v, T_accel, L) := block(
    [t1, t2, total_time, time_for_mouth2],
    /* 口2の経路: 加速→等速→減速→戻る */
    t1: T_accel, /* 加速時間 */
    t2: L/v,      /* 等速移動時間 */
    total_time: 2·t1 + t2,
   
   /* 特殊相対論的効果 */
    time_for_mouth2: total_time / lorentz_factor(v),
   
   /* 時間差 */
    time_for_mouth2 − total_time
);
